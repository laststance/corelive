#### Usage

<Source language="tsx" code={`import { AspectRatio } from '@/components/ui/aspect-ratio'

export default function Example() {
  return (
    <div className="w-[360px] border rounded-md overflow-hidden">
      <AspectRatio ratio={16/9}>
        <img src="https://picsum.photos/800/450" alt="16:9" className="h-full w-full object-cover" />
      </AspectRatio>
    </div>
  )
}`} />
